def grad_descent(X, y, T, alpha):
    m, n = X.shape
    
    #fill in theta and f with zeros
    theta = np.zeros(n)
    f = np.zeros(T)
    
    for i in range(T):
        #calculate the loss
        f[i] = np.linalg.norm(X.dot(theta)-y, 1)
        #steepest ascent at f(theta)
        g = X.T.dot(np.sign(X.dot(theta)-y))
        #gradient step
        theta = theta - alpha*g
        
    return theta, f

def grad_descent_log(X, y, T, alpha):
    m, n = X.shape
    
    #fill in theta and f with zeros
    theta = np.zeros(n)
    f = np.zeros(T)
    
    for i in range(T):
        #calculate the loss
        f[i] = np.linalg.norm(X.dot(theta)-y, 1)
        #steepest ascent at f(theta)
        fbk = X.dot(theta)
        pbk = [np.exp(j)/(1+np.exp(j)) for j in fbk]
        g = X.T.dot([j-k for j, k in zip(y,pbk)])
        #gradient step
        theta = theta + alpha*g
        
    return theta, f
